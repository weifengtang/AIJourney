## 三、核心架构设计

### 3.1 采集器插件机制（扩展性设计）

#### 3.1.1 采集器基类

所有采集器必须继承 `BaseCollector` 基类：

```python
# collectors/base.py
from abc import ABC, abstractmethod
from pathlib import Path
from typing import Optional
import json
from datetime import datetime

class BaseCollector(ABC):
    """采集器基类"""
    
    def __init__(self, name: str, output_subdir: str):
        """
        Args:
            name: 采集器名称（如 "Claude Code"）
            output_subdir: 输出子目录（如 "03_ClaudeCode"）
        """
        self.name = name
        self.output_subdir = output_subdir
        self.sessions = []  # 存储采集的会话数据
    
    @abstractmethod
    def collect(self, target_date: datetime) -> bool:
        """
        采集数据的核心方法（子类必须实现）
        
        Args:
            target_date: 目标日期
        
        Returns:
            成功返回 True，失败返回 False
        """
        pass
    
    def save(self, output_dir: Path) -> bool:
        """
        保存采集结果（JSON + Markdown 双格式）
        
        Args:
            output_dir: 输出目录（如 .../2026-03-25/03_ClaudeCode/）
        
        Returns:
            成功返回 True，失败返回 False
        """
        # 创建输出目录
        collector_dir = output_dir / self.output_subdir
        collector_dir.mkdir(parents=True, exist_ok=True)
        
        # 保存 JSON 格式
        json_file = collector_dir / "sessions.json"
        with open(json_file, 'w', encoding='utf-8') as f:
            json.dump(self.sessions, f, ensure_ascii=False, indent=2)
        
        # 保存 Markdown 格式
        md_file = collector_dir / f"{self.name}会话记录.md"
        md_content = self._generate_markdown()
        with open(md_file, 'w', encoding='utf-8') as f:
            f.write(md_content)
        
        return True
    
    @abstractmethod
    def _generate_markdown(self) -> str:
        """
        生成 Markdown 格式内容（子类实现）
        
        Returns:
            Markdown 格式的会话记录
        """
        pass
    
    def get_summary(self) -> dict:
        """
        获取采集结果摘要（供 main.py 汇总用）
        
        Returns:
            摘要信息字典
        """
        return {
            'collector': self.name,
            'session_count': len(self.sessions),
            'output_subdir': self.output_subdir
        }
```

#### 3.1.2 插件注册机制

```python
# collectors/__init__.py
from .base import BaseCollector
from typing import Dict, Type

# 插件注册表
COLLECTOR_REGISTRY: Dict[str, Type[BaseCollector]] = {}

def register_collector(name: str):
    """
    采集器注册装饰器
    
    使用示例：
    @register_collector('claude_code')
    class ClaudeCodeCollector(BaseCollector):
        ...
    """
    def decorator(cls: Type[BaseCollector]):
        COLLECTOR_REGISTRY[name] = cls
        return cls
    return decorator

def get_collector(name: str) -> Optional[Type[BaseCollector]]:
    """获取已注册的采集器类"""
    return COLLECTOR_REGISTRY.get(name)

def get_all_collectors() -> Dict[str, Type[BaseCollector]]:
    """获取所有已注册的采集器"""
    return COLLECTOR_REGISTRY.copy()
```

## 五、主程序流程（main.py）

```python
from datetime import datetime
from pathlib import Path
import logging
from config import Config
from collectors import get_all_collectors

def main():
    """主程序入口"""
    
    # 1. 初始化日志
    setup_logging()
    logger = logging.getLogger(__name__)
    
    # 2. 获取今天日期
    today = datetime.now()
    date_str = today.strftime(Config.DATE_FORMAT)
    logger.info(f"开始采集 {date_str} 的工作会话数据")
    
    # 3. 创建输出目录
    output_dir = Config.OUTPUT_BASE_DIR / date_str
    output_dir.mkdir(parents=True, exist_ok=True)
    logger.info(f"输出目录: {output_dir}")
    
    # 4. 按优先级加载并执行采集器
    collectors = get_all_collectors()
    summaries = []
    
    # 按优先级排序
    enabled_collectors = [
        (name, cls) for name, cls in collectors.items()
        if Config.COLLECTORS.get(name, {}).get('enabled', False)
    ]
    enabled_collectors.sort(
        key=lambda x: Config.COLLECTORS[x[0]].get('priority', 999)
    )
    
    for name, collector_cls in enabled_collectors:
        try:
            logger.info(f"执行采集器: {name}")
            collector = collector_cls()
            
            # 采集数据
            if collector.collect(today):
                # 保存结果
                collector.save(output_dir)
                summaries.append(collector.get_summary())
                logger.info(f"采集器 {name} 执行成功")
            else:
                logger.warning(f"采集器 {name} 未采集到数据")
                
        except Exception as e:
            # 容错：单个采集器失败不影响其他采集器
            logger.error(f"采集器 {name} 执行失败: {e}", exc_info=True)
            continue
    
    # 5. 生成今日工作总结
    if summaries:
        generate_daily_report(output_dir, summaries, today)
        logger.info(f"日报生成完成: {output_dir / '今日工作总结.md'}")
    else:
        logger.warning("所有采集器均未采集到数据")
    
    logger.info("采集任务完成")

if __name__ == "__main__":
    main()
```

## 六、容错与日志机制

### 6.1 容错策略

1. **采集器隔离**：单个采集器失败不影响其他采集器
2. **异常捕获**：所有采集器执行都在 try-except 中
3. **日志记录**：记录详细的错误信息和堆栈
4. **优雅降级**：部分数据缺失时仍可生成日报
