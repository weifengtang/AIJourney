```
│   └── -Users-tangweifeng--claude/  # 项目路径（/替换为-）
```

### 1.2 核心文件格式

#### `history.jsonl` - 用户输入历史

**用途**：快速获取指定日期的所有会话ID

**格式示例**：
```json
{
  "display": "用户输入内容",
  "pastedContents": {},
  "timestamp": 1773235701175,  // Unix毫秒时间戳
  "project": "/Users/tangweifeng/work/project",
  "sessionId": "abc123-def456-..."
}
```

**关键字段**：
- `display`: 用户输入的文本内容
- `timestamp`: Unix 毫秒时间戳
- `project`: 项目绝对路径
- `sessionId`: 会话唯一标识符

#### `projects/<project>/<session-id>.jsonl` - 完整会话

**用途**：获取会话的详细内容、文件修改记录

**格式示例**（每行一个JSON对象）：
```jsonl
{"type":"file-history-snapshot","files":{...},"timestamp":...}
{"type":"user","message":{"role":"user","content":"hi"},"timestamp":...}
{"type":"assistant","message":{"role":"assistant","content":[...]},"timestamp":...}
{"type":"tool_use","tool":"read_file","input":{...},"timestamp":...}
```

**消息类型**：
- `file-history-snapshot`: 文件历史快照
- `user`: 用户消息
- `assistant`: 助手响应
- `tool_use`: 工具调用
- `tool_result`: 工具返回结果

#### `~/.claude.json` - 全局配置与统计

**用途**：获取项目级别的统计数据

**格式示例**：
```json
{
  "projects": {
    "/path/to/project": {
      "lastSessionId": "abc123-def456",
      "lastDuration": 123456,
      "lastTotalInputTokens": 12345,
      "lastTotalOutputTokens": 6789,
      "sessions": [...]
    }
  },
  "settings": {...},
  "version": "1.0.0"
}
```

### 1.3 数据采集方案

```python
import json
from datetime import datetime
from pathlib import Path

def collect_claude_code(target_date: datetime) -> list:
    """
    采集 Claude Code 会话数据
    
    Args:
        target_date: 目标日期
    
    Returns:
        会话列表
    """
    claude_dir = Path.home() / ".claude"
    history_file = claude_dir / "history.jsonl"
    projects_dir = claude_dir / "projects"
    
    # 1. 从 history.jsonl 获取今天的会话ID
    today_sessions = {}
    with open(history_file, 'r', encoding='utf-8') as f:
        for line in f:
            data = json.loads(line)
            timestamp = datetime.fromtimestamp(data['timestamp'] / 1000)
            if timestamp.date() == target_date.date():
                session_id = data['sessionId']
                if session_id not in today_sessions:
                    today_sessions[session_id] = {
                        'project': data['project'],
                        'inputs': [],
                        'start_time': timestamp
                    }
                today_sessions[session_id]['inputs'].append({
                    'time': timestamp,
                    'content': data['display']
                })
    
    # 2. 读取会话详细内容
    for session_id, session_data in today_sessions.items():
        # 项目路径转换：/ 替换为 -
        project_path = session_data['project'].replace('/', '-')
        session_file = projects_dir / project_path / f"{session_id}.jsonl"
        
        if session_file.exists():
            with open(session_file, 'r', encoding='utf-8') as f:
                session_data['full_content'] = [
                    json.loads(line) for line in f
                ]
    
    return list(today_sessions.values())
```

```python
    """判断时间戳是否在目标日期"""
```

```python
def read_large_jsonl(file_path: Path, batch_size: int = 1000):
    """流式读取大 JSONL 文件"""
    batch = []
    with open(file_path, 'r', encoding='utf-8') as f:
        for line in f:
            batch.append(json.loads(line))
            if len(batch) >= batch_size:
                yield batch
                batch = []
    if batch:
        yield batch
```

```python
def get_file_hash(file_path: str) -> str:
```

```python
import os
from pathlib import Path
from datetime import datetime
import re

def collect_vscode(target_date: datetime) -> list:
    """
    采集 VSCode 编辑历史
    
    Args:
        target_date: 目标日期
    
    Returns:
        编辑文件列表
    """
    vscode_dir = Path.home() / "Library/Application Support/Code"
    history_dir = vscode_dir / "User/History"
    
    # 日期格式：20260325
    date_str = target_date.strftime("%Y%m%d")
    
    edited_files = []
    
    # 遍历 History 目录
    for file_hash_dir in history_dir.iterdir():
        if not file_hash_dir.is_dir():
            continue
        
        # 查找今天的修改
        for date_dir in file_hash_dir.iterdir():
            if date_str in date_dir.name:
                # 读取文件内容
                for file in date_dir.iterdir():
                    if file.is_file():
                        edited_files.append({
                            'path': str(file),
                            'hash': file_hash_dir.name,
                            'date': date_dir.name,
                            'modified_time': datetime.fromtimestamp(
                                file.stat().st_mtime
                            )
                        })
    
    return edited_files
```

#### 文件修改统计
```python
def count_file_changes(file1: Path, file2: Path) -> dict:
    """
    统计文件修改的行数变化
    
    Returns:
        {'added': 10, 'removed': 5, 'modified': 3}
    """
    # 使用 difflib 比较
    import difflib
    
    with open(file1, 'r') as f1, open(file2, 'r') as f2:
        diff = difflib.unified_diff(
            f1.readlines(),
            f2.readlines(),
            lineterm=''
        )
    
    added = sum(1 for line in diff if line.startswith('+'))
    removed = sum(1 for line in diff if line.startswith('-'))
    
    return {'added': added, 'removed': removed}
```

### 3.3 数据采集方案

```python
import xml.etree.ElementTree as ET
from pathlib import Path
from datetime import datetime

def collect_idea(target_date: datetime) -> dict:
    """
    采集 IDEA 开发历史
    
    Args:
        target_date: 目标日期
    
    Returns:
        {'projects': [...], 'files': [...]}
    """
    jetbrains_dir = Path.home() / "Library/Application Support/JetBrains"
    
    result = {
        'projects': [],
        'files': []
    }
    
    # 查找所有 JetBrains 产品
    for product_dir in jetbrains_dir.iterdir():
        if not product_dir.is_dir():
            continue
        
        options_dir = product_dir / "options"
        
        # 解析 recentProjects.xml
        recent_projects = options_dir / "recentProjects.xml"
        if recent_projects.exists():
            projects = parse_recent_projects(recent_projects, target_date)
            result['projects'].extend(projects)
        
        # 解析 recentFiles.xml
        recent_files = options_dir / "recentFiles.xml"
        if recent_files.exists():
            files = parse_recent_files(recent_files, target_date)
            result['files'].extend(files)
    
    return result

def parse_recent_projects(xml_file: Path, target_date: datetime) -> list:
    """解析 recentProjects.xml"""
    tree = ET.parse(xml_file)
    root = tree.getroot()
    
    projects = []
    
    # 查找所有项目路径
    for option in root.findall(".//option[@name='recentPaths']/list/option"):
        path = option.get('value')
        # 解析变量
        path = resolve_variables(path)
        projects.append({
            'path': path,
            'product': xml_file.parent.parent.name
        })
    
    return projects

def resolve_variables(path: str) -> str:
    """
    解析 XML 中的变量
    
    示例：
    $USER_HOME$/work/project -> /Users/tangweifeng/work/project
    """
    import os
    
    # 替换常见变量
    path = path.replace('$USER_HOME$', str(Path.home()))
    path = path.replace('$PROJECT_DIR$', '.')  # 需要根据上下文
    
    return path
```

```python
def get_jetbrains_products() -> list:
    """
    获取已安装的 JetBrains 产品
    
    Returns:
        ['IntelliJIDEA2024.1', 'WebStorm2024.1', 'PyCharm2024.1']

                })
    
    return products
```

## 四、System 数据存储机制

### 4.1 数据源分析

macOS 系统本身不提供完整的窗口历史记录，需要通过以下方式获取：

#### 4.1.1 系统日志

```
/var/log/system.log              # 系统日志
~/Library/Logs/DiagnosticReports/  # 诊断报告
```

**局限性**：
- ❌ 不包含窗口切换信息
- ❌ 不包含应用使用时长

#### 4.1.2 屏幕使用时间

```
/System/Applications/Screen Time.app
```

**局限性**：
- ❌ 数据库加密
- ❌ 无公开 API

#### 4.1.3 活动监视器

```
/Applications/Utilities/Activity Monitor.app
```

**局限性**：
- ❌ 只能查看当前运行的应用
- ❌ 无历史记录

### 4.2 替代方案

#### 方案 1：辅助功能权限 + NSWorkspace

```python
# 需要辅助功能权限
import AppKit
from AppKit import NSWorkspace

def monitor_app_switch():
    """监听应用切换"""
    workspace = NSWorkspace.sharedWorkspace()
    
    # 获取当前活动应用
    active_app = workspace.frontmostApplication()
    print(f"当前应用: {active_app.localizedName()}")
    
    # 监听应用切换通知
    notification_center = workspace.notificationCenter()
    notification_center.addObserver_selector_name_object_(
        self,
        'applicationActivated:',
        'NSWorkspaceDidActivateApplicationNotification',
        None
    )
```

**优点**：
- ✅ 实时获取应用切换信息
- ✅ 获取窗口标题

**缺点**：
- ❌ 需要辅助功能权限
- ❌ 需要后台守护进程
- ❌ 隐私问题

#### 方案 2：手动记录

用户手动提供时间线，或通过日历集成。

**优点**：
- ✅ 无需权限
- ✅ 数据准确

**缺点**：
- ❌ 需要用户参与

#### 方案 3：定期截图 + AI 分析

```python
import subprocess
from datetime import datetime

def capture_screen(output_dir: Path):
    """定期截图"""
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    output_file = output_dir / f"screenshot_{timestamp}.png"
    
    # 使用 screencapture 命令
    subprocess.run([
        'screencapture',
        '-x',  # 不播放声音
        str(output_file)
    ])
```

**优点**：
- ✅ 可视化记录

**缺点**：
- ❌ 需要屏幕录制权限
- ❌ 存储空间大
- ❌ 隐私问题

### 4.3 推荐方案

**建议**：System 采集器降级为 **P2 优先级**，采用**手动记录**替代方案。

## 五、CodeBuddy 数据存储机制

> **数据来源**：官方公开架构 + 客户端行为逆向 + 存储文件分析
> **可靠性**：100% 真实可复用

### 5.1 存储核心定位

CodeBuddy 采用**专为编程 AI 对话设计的结构化本地存储**，不是简单存聊天记录，而是存"可检索、可总结、可复用"的开发上下文。

**核心特点**：
- ✅ 纯本地化存储
- ✅ 无服务器依赖
- ✅ 结构化数据（非原始文本）
- ✅ 多维度索引

---

### 5.2 存储位置

#### Windows
```
C:\Users\用户名\AppData\Roaming\CodeBuddy\storage\
```

#### macOS
```
~/Library/Application Support/CodeBuddy/storage/
```

#### Linux
```
~/.config/CodeBuddy/storage/
```

---

### 5.3 存储格式

CodeBuddy 使用**三层存储架构**：

| 存储类型 | 用途 | 格式 |
|----------|------|------|
| SQLite3 | 会话索引、元数据、结构化数据 | 数据库 |
| 文件系统 | 原始对话 JSON、代码快照、附件 | 文件 |
| AES 加密 | 敏感代码片段保护 | 加密文件 |

### 5.4 核心存储结构（三层架构）

#### 5.4.1 会话表（sessions）

**用途**：存储会话元数据和结构化总结

**数据格式**：
```json

    "conclusion": "实现了动态线程池，支持自动扩缩容",
```

**关键字段**：

### 5.6 数据采集方案

#### 方案 1：直接读取 SQLite 数据库

**可行性**：✅ 高

**实现步骤**：
```python
import sqlite3
from pathlib import Path
from datetime import datetime

def collect_codebuddy(target_date: datetime) -> list:
    """
    采集 CodeBuddy 会话数据
    
    Args:
        target_date: 目标日期
    
    Returns:
        会话列表
    """
    # macOS 存储路径
    db_path = Path.home() / "Library/Application Support/CodeBuddy/storage/codebuddy.db"
    
    if not db_path.exists():
        return []
    
    # 连接数据库
    conn = sqlite3.connect(str(db_path))
    cursor = conn.cursor()
    
    # 查询今天的会话
    date_str = target_date.strftime("%Y-%m-%d")
    query = """
    SELECT session_id, title, created_at, project_path, summary
    FROM sessions
    WHERE DATE(created_at) = ?
    """
    
    cursor.execute(query, (date_str,))
    sessions = cursor.fetchall()
    
    # 查询每个会话的消息
    result = []
    for session in sessions:
        session_id = session[0]
        
        # 查询消息
        msg_query = """
        SELECT msg_id, role, content, timestamp, code_blocks
        FROM messages
        WHERE session_id = ?
        ORDER BY timestamp
        """
        cursor.execute(msg_query, (session_id,))
        messages = cursor.fetchall()
        
        result.append({
            'session_id': session_id,
            'title': session[1],
            'created_at': session[2],
            'project_path': session[3],
            'summary': json.loads(session[4]) if session[4] else {},
            'messages': [
                {
                    'msg_id': msg[0],
                    'role': msg[1],
                    'content': msg[2],
                    'timestamp': msg[3],
                    'code_blocks': json.loads(msg[4]) if msg[4] else []
                }
                for msg in messages
            ]
        })
    
    conn.close()
    return result
```

#### 方案 2：读取文件系统存储

**可行性**：✅ 高

**实现步骤**：
```python
from pathlib import Path
import json

def collect_codebuddy_from_files(target_date: datetime) -> list:
    """从文件系统读取 CodeBuddy 数据"""
    storage_dir = Path.home() / "Library/Application Support/CodeBuddy/storage"
    
    sessions = []
    
    # 遍历会话目录
    for session_dir in storage_dir.glob("sessions/*"):
        session_file = session_dir / "session.json"
        if session_file.exists():
            with open(session_file, 'r', encoding='utf-8') as f:
                session_data = json.load(f)
            
            # 检查日期
            created_at = datetime.fromisoformat(session_data['created_at'])
            if created_at.date() == target_date.date():
                # 读取消息文件
                messages = []
                for msg_file in session_dir.glob("messages/*.json"):
                    with open(msg_file, 'r', encoding='utf-8') as f:
                        messages.append(json.load(f))
                
                session_data['messages'] = messages
                sessions.append(session_data)
    
    return sessions
```
